---
title: "highlight()"
description: "Create a highlight.js syntax highlighting plugin that highlights fenced code blocks via HAST transforms using lowlight."
---

```ts
import { highlight } from "@unifast/highlight";
```

## Signature

```ts
function highlight(): UnifastPlugin
```

## Parameters

None.

## Usage

```ts
import { compile } from "@unifast/node";
import { highlight } from "@unifast/highlight";

const md = `
\`\`\`typescript
const greeting: string = "Hello";
console.log(greeting);
\`\`\`
`;

const result = compile(md, {
  plugins: [highlight()],
});

console.log(result.output);
// <pre><code class="language-typescript hljs">
//   <span class="hljs-keyword">const</span> ...
// </code></pre>
```

## Examples

### With multiple languages

```ts
import { compile } from "@unifast/node";
import { highlight } from "@unifast/highlight";

const md = `
\`\`\`python
def hello():
    print("Hello, world!")
\`\`\`

\`\`\`css
.container {
  display: flex;
}
\`\`\`
`;

const result = compile(md, { plugins: [highlight()] });
// Both code blocks are highlighted with appropriate highlight.js classes
```

### With other plugins

```ts
import { compile, gfm, frontmatter } from "@unifast/node";
import { highlight } from "@unifast/highlight";

const result = compile(source, {
  plugins: [gfm(), frontmatter(), highlight()],
});
```
