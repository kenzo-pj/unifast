```ts
import { cjk } from "@unifast/node";
```

## Signature

```ts
function cjk(): UnifastPlugin
```

## Parameters

None.

## Usage

```ts
import { compile, cjk } from "@unifast/node";

const md = `
今日は天気が
とても良いです。
`;

const result = compile(md, { plugins: [cjk()] });
console.log(result.output);
// <p>今日は天気がとても良いです。</p>
```
