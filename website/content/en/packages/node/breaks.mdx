## Signature

```ts
function breaks(): UnifastPlugin
```

## Parameters

None.

## Usage

```ts
import { compile, breaks } from "@unifast/node";

const md = `This is line one.
This is line two.
This is line three.`;

const result = compile(md, {
  plugins: [breaks()],
});
// Each newline becomes a <br> element
```
