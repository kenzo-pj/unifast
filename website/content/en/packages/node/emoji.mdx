```ts
import { emoji } from "@unifast/node";
```

### Signature

```ts
function emoji(): UnifastPlugin
```

### Parameters

None.

### Returns

`UnifastPlugin`

## Examples

<Example data={example} />

## Usage

```ts
import { compile } from "@unifast/node";
import { emoji } from "@unifast/node";

const md = `This is great :thumbsup: and I'm happy :smile:!`;

const result = compile(md, {
  plugins: [emoji()],
});
// :thumbsup: → 👍, :smile: → 😄
```
